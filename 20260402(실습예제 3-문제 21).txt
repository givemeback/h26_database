(1)테이블을 생성하는 CREATE 문과 데이터를 삽입하는 INSERT문을 작성하시오. 테이블의  데이터  타입은  임의로  정하고,  데이터는  다음  질의의  결과가  나오도록 삽입한다.

CREATE TABLE Salesperson (
    name VARCHAR2(50) PRIMARY KEY,
    age NUMBER,
    salary NUMBER);

CREATE TABLE Customer (
    name VARCHAR2(50) PRIMARY KEY,
    city VARCHAR2(50),
    industrytype VARCHAR2(50));

CREATE TABLE "Order" (
    num NUMBER PRIMARY KEY,
    custname VARCHAR2(50),
    salesperson VARCHAR2(50),
    amount NUMBER,
    CONSTRAINT FK_ORDER_CUSTOMER FOREIGN KEY (custname) REFERENCES Customer(name),
    CONSTRAINT FK_ORDER_SALES FOREIGN KEY (salesperson) REFERENCES Salesperson(name));


COMMIT;
